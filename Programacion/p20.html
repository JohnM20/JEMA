Generación de un PDF

Usaremos ReportLab para generar un PDF con los resultados del análisis.

from reportlab.lib.pagesizes import letter
from reportlab.pdfgen import canvas

# Crear un PDF
pdf = canvas.Canvas("analisis_datos.pdf", pagesize=letter)
pdf.drawString(100, 750, "Análisis de Datos")
pdf.drawString(100, 730, "Resultados del análisis en Excel.")
pdf.save()
